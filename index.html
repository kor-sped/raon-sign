let themeIdx = 0;

let currentPage = 'home';


function applyTheme() {

  const t = THEMES[themeIdx];

  document.body.style.background = t.bg;

  document.body.style.color = t.text;

  document.getElementById('header').style.background = t.primary;

  document.getElementById('theme-btn').style.background = 'rgba(255,255,255,0.25)';

  const nav = document.getElementById('nav');

  nav.style.borderBottomColor = t.primary;

  nav.style.background = t.card;

  document.querySelectorAll('#nav button').forEach(b => {

    b.style.background = b.classList.contains('active') ? t.primary : t.card;

    b.style.color = b.classList.contains('active') ? '#fff' : t.text;

  });

  document.getElementById('theme-sheet').style.background = t.card;

  document.getElementById('theme-sheet').style.color = t.text;

  buildThemeGrid();

}


function buildThemeGrid() {

  const grid = document.getElementById('theme-grid');

  grid.innerHTML = '';

  THEMES.forEach((th, i) => {

    const btn = document.createElement('button');

    btn.className = 'theme-btn' + (i === themeIdx ? ' current' : '');

    btn.style.background = th.primary;

    btn.style.color = '#fff';

    if (i === themeIdx) btn.style.boxShadow = `0 0 0 3px ${th.accent}`;

    btn.textContent = th.name;

    btn.onclick = () => { themeIdx = i; closeTheme(); applyTheme(); render(); };

    grid.appendChild(btn);

  });

}


function openTheme() { document.getElementById('theme-overlay').classList.add('open'); }

function closeTheme() { document.getElementById('theme-overlay').classList.remove('open'); }


function getYTId(url) {

  if (!url) return null;

  const m = url.match(/(?:youtu\.be\/|v=|embed\/)([^&?/]+)/);

  return m ? m[1] : null;

}

}

function goPage(page) {

  currentPage = page;

  document.querySelectorAll('#nav button').forEach((b, i) => {

    const pages = ['home','beginner','vocab','advanced','dict'];

    b.classList.toggle('active', pages[i] === page);

  });

  applyTheme();

  render();

}
